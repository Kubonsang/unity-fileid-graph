# v0.1 Lossless Block Parser Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a read-only Go CLI that parses Unity YAML files into ordered lossless blocks and prints deterministic `blocks` output for `prefab`, `scene`, `asset`, and `mat` namespaces.

**Architecture:** Keep v0.1 intentionally narrow: `cmd/uyaml` delegates to `internal/cli`, which validates the command shape and calls `internal/parser` to produce a `core.ParseResult`. The parser is a header-driven block splitter that preserves `HeaderRaw`, `BodyRaw`, block order, newline style, and trailer text without attempting any YAML object mapping.

**Tech Stack:** Go 1.22, standard library only, `go test`, table-driven tests, golden-file CLI assertions.

```go
package parser

import (
	"fmt"
	"regexp"
	"strconv"
)

type headerMeta struct {
	ClassID    int
	FileID     int64
	IsStripped bool
}

var headerPattern = regexp.MustCompile(`^--- !u!(\d+) &(-?\d+)( stripped)?(?:\r?\n)$`)

func parseHeader(header string) (headerMeta, error) {
	matches := headerPattern.FindStringSubmatch(header)
	if matches == nil {
		return headerMeta{}, fmt.Errorf("invalid Unity header: %q", header)
	}

	classID, err := strconv.Atoi(matches[1])
	if err != nil {
		return headerMeta{}, fmt.Errorf("parse class id: %w", err)
	}
	fileID, err := strconv.ParseInt(matches[2], 10, 64)
	if err != nil {
		return headerMeta{}, fmt.Errorf("parse file id: %w", err)
	}

	return headerMeta{
		ClassID:    classID,
		FileID:     fileID,
		IsStripped: matches[3] == " stripped",
	}, nil
}
```

```go
func Parse(input []byte) (*core.ParseResult, error) {
	src := string(input)
	result := &core.ParseResult{}

	offsets := headerPattern.FindAllStringIndex(src, -1)
	if len(offsets) == 0 {
		result.PreambleRaw = src
		return result, nil
	}

	result.PreambleRaw = src[:offsets[0][0]]
	result.Blocks = make([]*core.Block, 0, len(offsets))

	for i, offset := range offsets {
		headerStart := offset[0]
		headerEnd := offset[1]
		nextStart := len(src)
		if i+1 < len(offsets) {
			nextStart = offsets[i+1][0]
		}

		headerRaw := src[headerStart:headerEnd]
		meta, err := parseHeader(headerRaw)
		if err != nil {
			return nil, err
		}

		block := &core.Block{
			Index:      i,
			ClassID:    meta.ClassID,
			FileID:     meta.FileID,
			HeaderRaw:  headerRaw,
			BodyRaw:    src[headerEnd:nextStart],
			IsStripped: meta.IsStripped,
		}
		result.Blocks = append(result.Blocks, block)
	}

	if last := result.Blocks[len(result.Blocks)-1]; last != nil {
		result.TrailerRaw = trailingDocumentMarker(last.BodyRaw)
		if result.TrailerRaw != "" {
			last.BodyRaw = last.BodyRaw[:len(last.BodyRaw)-len(result.TrailerRaw)]
		}
	}

	return result, nil
}

func trailingDocumentMarker(body string) string {
	if len(body) >= 4 && body[len(body)-4:] == "...\n" {
		return "...\n"
	}
	if len(body) >= 5 && body[len(body)-5:] == "...\r\n" {
		return "...\r\n"
	}
	return ""
}
```

```go
package cli

import (
	"fmt"
	"io"
	"os"

	"unity-fileid-graph/internal/parser"
)

func Run(args []string, stdout, stderr io.Writer) int {
	if len(args) != 3 {
		writeUsage(stderr)
		return 2
	}
	if _, ok := validNamespaces[args[0]]; !ok {
		writeUsage(stderr)
		return 2
	}
	if args[1] != "blocks" {
		writeUsage(stderr)
		return 2
	}

	input, err := os.ReadFile(args[2])
	if err != nil {
		_, _ = fmt.Fprintf(stderr, "read file: %v\n", err)
		return 1
	}
	result, err := parser.Parse(input)
	if err != nil {
		_, _ = fmt.Fprintf(stderr, "parse file: %v\n", err)
		return 1
	}
	for _, block := range result.Blocks {
		stripped := 0
		if block.IsStripped {
			stripped = 1
		}
		_, _ = fmt.Fprintf(stdout, "BLOCK index=%d class_id=%d file_id=%d stripped=%d\n",
			block.Index, block.ClassID, block.FileID, stripped)
	}
	return 0
}
```
